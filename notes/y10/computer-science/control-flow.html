Control Flow

If else

if (condition) {
    // code to execute if condition is true
} else {
    // code to execute if condition is false
}

Example

int x = 5;
if (x == 5) {
    cout << "x is 5" << endl;
} else if (x % 2) { // any non-zero number is true
    cout << "x is odd" << endl;
} else {
    cout << "x is even" << endl;
}
Converts to true Converts to false
Literally everything 0
null
false
"" (empty string)

Switch

int x = 2;

switch (x) {
    case 1:
        cout << "x is 1" << endl;
        break;
    case 2:
        cout << "x is 2" << endl;
        break;
    case 3:
        cout << "x is 3" << endl;
        break;
    default:
        cout << "x is not 1, 2, or 3" << endl;
}

Note

You cannot switch on strings.

Loops

// while loop
while (condition) {
    // code to execute
}

// for loop
for (/* do before loop */; /* condition */; /* do after each iteration */) {
    // code to execute
}

// do while loop - executes at least once
do {
    // code to execute
} while (condition);
