Libraries

To print out hello world, we need to include the iostream library 

#include <iostream>

using namespace std;

int main()
{
    // without bringing `std` to namespace, we have to specify `std::cout`
    // std::cout << "Hello World!" << std::endl;

    cout << "Hello World!" << endl; // a semicolon as line break
    return 0;
}

x = 5;

x = 5;
x += 10; // x = x + 10 (x is 15)
x /= 3; // x = x / 3 (x is 5)

z = x % 2; // remainder of x divided by 2 (z is 1)
